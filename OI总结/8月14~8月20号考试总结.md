``` cpp
inline ll query(int x,int L,int R){
	if(l(x)>=L&&r(x)<=R){return sum(x);}
	push_down(x);
	int mid=(L+R)>>1;
	ll ans=0;
	if(mid>=L){ans+=query(lc,L,R);}
	if(mid<R){ans+=query(rc,L,R);}
	return ans;
}
```
AC代码：
```cpp
inline ll query(int x,int L,int R){
	if(l(x)>=L&&r(x)<=R){return sum(x);}
	push_down(x);
	int mid=(l(x)+r(x))>>1;
	ll ans=0;
	if(mid>=L){ans+=query(lc,L,R);}
	if(mid<R){ans+=query(rc,L,R);}
	return ans;
}
```
